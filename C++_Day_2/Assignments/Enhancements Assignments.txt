6) write a function to swap 2 given numbers using pointers.
==>
#include <iostream>
#include "First.h"
using namespace std;
void swap(int* a,int* b) {
	int temp = *a;
	*a = *b;
	*b = temp;
}
int main()
{
	int a = 10;
	int b = 20;

	cout << a << " " << b << endl;

	swap(&a, &b);
	cout << a << " " << b;
}

9)write a function which takes 3 arguments
variable, pointer and a reference.(any datatype)
invoke this function and show how "call by value","call by address" and "call by reference" will work.
==>
#include <iostream>
#include "First.h"
using namespace std;
//void disp(int a, int b, int c) {
//	cout << a << " " << b << " " << c << endl;
//}
void disp(int* a,int* b,int* c) {
	cout << *a << " " << *b << " " << *c << endl;
}
void disp(const int& a,const int& b,const int& c) {
	cout << a << " " << b << " " << c << endl;
}
int main()
{
	int a = 10;
	int b = 20;
	int c = 30;

	disp(10, 20, 30); //call by value
	disp(&a, &b, &c); //call by address
	disp(a, b, c); //call by reference 
}
